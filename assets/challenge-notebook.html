# applying the same transformations to the test data
def transforms(dataframe):
    # filling missing data points for bool features
    dataframe['balcony'] = dataframe['balcony'].fillna(False)
    dataframe['balcony'] = dataframe['balcony'].astype(int)
    
    dataframe['elevator'] = dataframe['elevator'].fillna(False)
    dataframe['elevator'] = dataframe['elevator'].astype(int)
    
    dataframe['garden'] = dataframe['garden'].fillna(False)
    dataframe['garden'] = dataframe['garden'].astype(int)
    
    # filling in missing conditions data
    dataframe['conditions'] = dataframe['conditions'].fillna('unknown')
    
    # drop energy efficiency, expenses, construction year, id
    global index
    index = dataframe['id']
    dataframe.drop('id', axis=1, inplace=True)
    dataframe.drop('energy_efficiency', axis=1, inplace=True)
    dataframe.drop('expenses', axis=1, inplace=True)
    dataframe.drop('construction_year', axis=1, inplace=True)
    dataframe.drop('total_floors', axis=1, inplace=True)
    
    #filling in missing values
#     dataframe['latitude'] = dataframe['latitude'].fillna(0)
#     dataframe['longitude'] = dataframe['longitude'].fillna(0)
    dataframe['proximity_to_center'] = dataframe['proximity_to_center'].fillna(0)
    
    my_imputer = SimpleImputer(strategy='median')
    dataframe[['balcony', 'latitude', 'longitude', 'floor',
       'garden', 'n_bathrooms', 'elevator', 'n_rooms',
       'proximity_to_center', 'surface']] = my_imputer.fit_transform(dataframe[['balcony', 'latitude', 'longitude', 'floor',
       'garden', 'n_bathrooms', 'elevator', 'n_rooms',
       'proximity_to_center', 'surface']])
    
    #one-hot encoding condiitons
    dataframe = pd.get_dummies(dataframe, columns = ['conditions'])
    
    #Cities + one-hot encoding
    results = rg.search(list(zip(dataframe['latitude'], dataframe['longitude'])))
    cities = [result['admin2'] for result in results]
    
    dataframe['city'] = cities
    dataframe = pd.get_dummies(dataframe, columns = ['city'])
